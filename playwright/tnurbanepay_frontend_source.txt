        (function () {
            function parseIso(s) {
                if (!s) return null;
                try {
                    var d = new Date(s);
                    return isNaN(d.getTime()) ? null : d;
                } catch (e) { return null; }
            }
            function formatSeconds(sec) { if (sec < 0) sec = 0; var m = Math.floor(sec / 60); var s = Math.floor(sec % 60); return (m < 10 ? "0" + m : m) + ":" + (s < 10 ? "0" + s : s); }

            var hfResend = document.getElementById('PageContent_hfResendTimestamp');
            var hfExpiry = document.getElementById('PageContent_hfOTPExpiry');
            var btnResend = document.getElementById('PageContent_btnResend');
            var lblCountdown = document.getElementById('PageContent_lblCountdown');
            var lblResendTimer = document.getElementById('PageContent_lblResendTimer');

            var resendTs = hfResend && hfResend.value ? parseIso(hfResend.value) : null;
            var expiryTs = hfExpiry && hfExpiry.value ? parseIso(hfExpiry.value) : null;
            var RESEND_COOLDOWN_SECONDS = 60;

            function updateUI() {
                var now = new Date();

                if (resendTs) {
                    var secondsSince = Math.floor((now - resendTs) / 1000);
                    var remainingResend = RESEND_COOLDOWN_SECONDS - secondsSince;
                    if (remainingResend > 0) {
                        if (btnResend) btnResend.disabled = true;
                        if (lblResendTimer) lblResendTimer.innerText = "Resend in " + formatSeconds(remainingResend);
                    } else {
                        if (btnResend) btnResend.disabled = false;
                        if (lblResendTimer) lblResendTimer.innerText = "";
                    }
                } else {
                    if (btnResend) btnResend.disabled = false;
                    if (lblResendTimer) lblResendTimer.innerText = "";
                }

                if (expiryTs) {
                    var remaining = Math.ceil((expiryTs - now) / 1000.0);
                    if (remaining > 0) {
                        if (lblCountdown) lblCountdown.innerText = "OTP expires in " + formatSeconds(remaining);
                    } else {
                        if (lblCountdown) lblCountdown.innerText = "OTP expired";
                        if (btnResend) btnResend.disabled = false;
                        expiryTs = null;
                        if (hfExpiry) hfExpiry.value = "";
                    }
                } else {
                    if (lblCountdown) lblCountdown.innerText = "";
                }
            }

            updateUI();
            setInterval(updateUI, 1000);
